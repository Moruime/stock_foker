## Task 1: 后端 - 新增对比基准 API

文件：`backend/app/routers/stock_router.py`

新增 `GET /api/stocks/{stock_code}/benchmark` 端点：
- 参数：`period`（默认 daily）、`days`（默认 120，近半年）
- 返回：个股 + 沪深300 + 上证指数的日收盘价序列 + 超额收益统计

```python
# 返回结构
{
  "dates": ["2026-01-02", ...],
  "stock": {
    "name": "贵州茅台",
    "close": [1800.0, ...],        # 原始收盘价
    "pct_change": [0, 0.5, -0.3, ...],  # 相对起点的累计涨跌幅%
  },
  "benchmarks": [
    {
      "code": "000300",
      "name": "沪深300",
      "close": [3900.0, ...],
      "pct_change": [0, 0.2, ...],
    },
    {
      "code": "000001",
      "name": "上证指数",
      "close": [3100.0, ...],
      "pct_change": [0, 0.3, ...],
    },
  ],
  "stats": {
    "stock_return": 12.5,          # 区间累计涨跌幅%
    "hs300_return": 5.2,
    "sh_return": 4.8,
    "excess_hs300": 7.3,           # 超额收益（vs 沪深300）
    "excess_sh": 7.7,              # 超额收益（vs 上证）
  }
}
```

实现逻辑：
- 复用 `get_kline_data()` 获取个股 K 线（已有缓存）
- 用同样函数获取 `000300`（沪深300）和 `000001`（上证指数）的 K 线
- 三组日期取交集对齐
- 计算累计涨跌幅：`pct = (close[i] / close[0] - 1) * 100`

文件：`backend/app/services/stock_service.py`

新增 `get_benchmark_comparison()` 函数封装计算逻辑。

## Task 2: 前端 - api.ts 和 types 扩展

文件：`frontend/src/services/api.ts`

```typescript
export interface BenchmarkData {
  dates: string[];
  stock: { name: string; close: number[]; pct_change: number[] };
  benchmarks: { code: string; name: string; close: number[]; pct_change: number[] }[];
  stats: {
    stock_return: number;
    hs300_return: number;
    sh_return: number;
    excess_hs300: number;
    excess_sh: number;
  };
}

export const getBenchmark = (stockCode: string, period?: string, days?: number) =>
  api.get<BenchmarkData>(`/stocks/${stockCode}/benchmark`, {
    params: { period, days },
  }).then(r => r.data);
```
